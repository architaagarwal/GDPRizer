SELECT * FROM CUSTOMER WHERE C_CUSTKEY = ds_id;
SELECT * FROM ORDERS WHERE O_CUSTKEY = ds_id;
SELECT * FROM NATION WHERE N_NATIONKEY IN (SELECT C_NATIONKEY FROM CUSTOMER WHERE C_CUSTKEY = ds_id);
SELECT * FROM REGION WHERE R_REGIONKEY IN (SELECT N_REGIONKEY FROM NATION WHERE N_NATIONKEY IN (SELECT C_NATIONKEY FROM CUSTOMER WHERE C_CUSTKEY = ds_id));
SELECT * FROM LINEITEM WHERE L_ORDERKEY IN (SELECT O_ORDERKEY FROM ORDERS WHERE O_CUSTKEY = ds_id);
SELECT * FROM PART WHERE P_PARTKEY IN (SELECT L_PARTKEY FROM LINEITEM WHERE L_ORDERKEY IN (SELECT O_ORDERKEY FROM ORDERS WHERE O_CUSTKEY = ds_id));
